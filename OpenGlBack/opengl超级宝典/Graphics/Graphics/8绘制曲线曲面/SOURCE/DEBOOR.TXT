   B样条曲线―deBoor分割算法

float deboor(degree,coeff,knot,u,i)
 float coeff[],knot[];
 float u;
 int degree,i;
{
 int k,j;
 float t1,t2;
 float coeffa[30];
 for(j=i-degree+1;j<=i+1;j++)
  coeffa[j]=coeff[j-i+degree-1];
 for(k=1;i<=degree;k++)
  forj=i+1;j>=i-degree+k+1;j--)
  {
   t1=(knot[j+degree-k]-u)/(knot[j+degree-k]-knot[j-1]);
   t2=1.0-t1;
   coeffa[j]=t1*coeffa[j-1]+t2*coeffa[j];
  }
 return (coeffa[i+1]);
}
